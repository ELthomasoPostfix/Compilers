for (int idx1 = 0; idx1 < 100; ++idx1) {
  if (idx1 % 2 == 0)
    continue;
}


int count = 0;
for (int i = 0; i < 100; ++i) {
  int rand_val;
  for (int j = 0; j < rand_val - i; ++j)
    if (j == rand_val) {
      count = count + 1;
      break;
    }
}


int rand_a;
if (rand_a < 0) rand_a = -1 * rand_a;

while ( rand_a-- ) {
    if (rand_a == 0)
      break;
    if (rand_a % 2 == 1)
        continue;
}
